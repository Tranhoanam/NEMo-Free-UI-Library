-- 2. Tạo Menu (Header)
local Window = Library:CreateWindow({
    Name = "NEMo Free UI Library"
})

-- 3. Tạo Tab (Ví dụ: Tab chính)

    Name = "Main"
})

    print("Nút đã được nhấn!")
end)

        -- Off
    end
end)

    -- Slider value
end)
